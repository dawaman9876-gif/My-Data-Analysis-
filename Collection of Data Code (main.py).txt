# Environmental Sensor Class: Handles DHT22 temperature and humidity readings with proper error handling.

# Concurrency: The ultrasonic sensor, camera monitoring and monitor_weather_conditions()  run in separate 'daemon threads'. Daemon Threads are designed to run in the background, providing services or 
               performing ancillary tasks without directly affecting the primary execution flow of the application so it means they continuously gather data in the 
               background without stopping the main robot from moving.

# Data Flow:
1.The monitor_obstacles() thread constantly updates the current_distance global variable. 
2.The monitor_environment() thread updates the latest_camera_data global variable every few seconds.

# Decision Making: The main_navigation_loop() (running in the main thread) checks these global variables and makes decisions based on priority:
1.Priority 1: Avoid obstacles (if current_distance < 20).
2.Priority 2: Spray if larvae are detected (elif latest_camera_data['detection_status']).
3.Priority 3: When temperature is 25-30°C and humidity >70% (ideal for mosquitoes), patrol intensity increases.
3.Priority 4: Default patrol behavior (else).

# Data Logging: Every sensor reading and significant navigation event is written to the auto_vector.log file with a timestamp using Python's logging module. 
                This creates a complete record of the mission for later analysis.*

# main.py
import threading
import time
import logging
from datetime import datetime
from motor_controller import MotorController
from ultrasonic_sensor import UltrasonicSensor
from camera_processor import CameraProcessor
from environmental_sensor import EnvironmentalSensor  # NEW IMPORT

# --- Setup Logging ---
logging.basicConfig(
    filename='auto_vector.log',
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger("AutoVectorCore")

# --- Initialize All Hardware Objects ---
print("Auto-Vector Booting Up...")
logger.info("Initializing All Hardware Components")

motor_controller = MotorController(ena_pin=18, in1_pin=17, in2_pin=27, enb_pin=24, in3_pin=22, in4_pin=23)
ultrasonic = UltrasonicSensor(trigger_pin=5, echo_pin=6)
camera = CameraProcessor()
env_sensor = EnvironmentalSensor()  # NEW: Environmental sensor

# --- Global Variables for Thread Communication ---
current_distance = 100
latest_camera_data = {"detection_status": False, "larva_count": 0}
current_temperature = None  # NEW
current_humidity = None     # NEW
is_running = True

# --- Thread 1: Ultrasonic Monitoring ---
def monitor_obstacles():
    """Continuously checks for obstacles."""
    global current_distance
    while is_running:
        dist = ultrasonic.get_distance()
        if 2 < dist < 400:
            current_distance = dist
            logger.info(f"OBSTACLE_DATA - Distance: {dist} cm")
        time.sleep(0.1)

# --- Thread 2: Camera Processing ---
def monitor_environment():
    """Periodically processes images for larva detection."""
    global latest_camera_data
    while is_running:
        data = camera.capture_and_process_frame()
        latest_camera_data = data
        logger.info(f"CAMERA_DATA - Larvae Detected: {data['detection_status']}, Count: {data['larva_count']}")
        time.sleep(3)

# --- NEW Thread 3: Environmental Monitoring ---
def monitor_weather_conditions():
    """Periodically reads temperature and humidity data."""
    global current_temperature, current_humidity
    while is_running:
        if env_sensor.read_data():  # Only update if read was successful
            current_temperature = env_sensor.get_temperature()
            current_humidity = env_sensor.get_humidity()
        time.sleep(30)  # Read every 30 seconds (environment changes slowly)

# --- Enhanced Navigation Logic with Environmental Awareness ---
def main_navigation_loop():
    """Core decision-making loop that uses all sensor data."""
    global is_running
    
    print("Starting Enhanced Navigation Loop. Press Ctrl+C to stop.")
    logger.info("NAVIGATION - Starting enhanced loop with environmental awareness")
    
    try:
        while is_running:
            # Log environmental conditions periodically
            if current_temperature is not None and current_humidity is not None:
                logger.info(f"ENV_CONDITIONS - Current: {current_temperature:.1f}C, {current_humidity:.1f}% RH")
            
            # --- ENVIRONMENT-BASED DECISION MAKING ---
            # Example: Adjust behavior based on optimal mosquito breeding conditions
            if (current_temperature is not None and current_humidity is not None):
                optimal_conditions = (current_temperature >= 25 and current_temperature <= 30 
                                     and current_humidity >= 70)
                
                if optimal_conditions:
                    logger.info("NAVIGATION - Optimal breeding conditions detected. Increasing patrol intensity.")
                    patrol_speed = 50  # Faster patrol in good conditions
                    patrol_duration = 10  # Patrol longer in good conditions
                else:
                    patrol_speed = 30  # Slower patrol otherwise
                    patrol_duration = 5  # Shorter patrol cycles
            else:
                patrol_speed = 40  # Default speed
                patrol_duration = 7  # Default duration
            
            # --- OBSTACLE AVOIDANCE (Highest Priority) ---
            if current_distance < 20:
                logger.warning("NAVIGATION - Obstacle detected! Avoiding...")
                motor_controller.stop()
                time.sleep(0.5)
                motor_controller.move_backward(40)
                time.sleep(1)
                motor_controller.turn_right(60)
                time.sleep(1.2)
                
            # --- LARVA DETECTION & ACTION ---
            elif latest_camera_data['detection_status']:
                logger.info("NAVIGATION - Target detected! Initiating spray sequence.")
                motor_controller.stop()
                
                # Enhanced spraying logic based on environment
                spray_intensity = 2  # seconds default
                if current_temperature > 28:  # Hotter temperatures might require less spray
                    spray_intensity = 1.5
                    logger.info("NAVIGATION - High temperature detected. Adjusting spray intensity.")
                
                time.sleep(spray_intensity)  # Simulate spraying
                logger.info("NAVIGATION - Spraying complete. Resuming patrol.")
                
                motor_controller.turn_left(40)
                time.sleep(0.5)
                
            # --- DEFAULT BEHAVIOR: Environment-Aware Patrol ---
            else:
                logger.info(f"NAVIGATION - Beginning patrol cycle (Speed: {patrol_speed}, Duration: {patrol_duration}s)")
                start_time = time.time()
                
                # Patrol for the determined duration
                while (time.time() - start_time) < patrol_duration and is_running:
                    if current_distance < 30:  # Check for obstacles during patrol
                        break
                    motor_controller.move_forward(patrol_speed)
                    time.sleep(0.1)
                
                # Change direction after patrol cycle
                motor_controller.turn_right(50)
                time.sleep(0.8)
            
            time.sleep(0.1)
            
    except KeyboardInterrupt:
        print("Shutdown signal received.")
    finally:
        is_running = False
        logger.info("NAVIGATION - Shutting down.")
        motor_controller.stop()
        motor_controller.cleanup()
        ultrasonic.cleanup()
        camera.cleanup()
        env_sensor.cleanup()  # NEW: Clean up environmental sensor
        print("Auto-Vector shutdown complete. Data logged to auto_vector.log")

# --- Start All Monitoring Threads ---
if __name__ == "__main__":
    # Create and start all threads
    obstacle_thread = threading.Thread(target=monitor_obstacles)
    camera_thread = threading.Thread(target=monitor_environment)
    weather_thread = threading.Thread(target=monitor_weather_conditions)  # NEW THREAD
    
    # Set as daemon threads (will exit when main program exits)
    obstacle_thread.daemon = True
    camera_thread.daemon = True
    weather_thread.daemon = True
    
    # Start all threads
    obstacle_thread.start()
    camera_thread.start()
    weather_thread.start()
    
    # Small delay to let all sensors initialize
    time.sleep(2)
    
    # Display initial environmental conditions
    if env_sensor.read_data():
        print(f"Initial Environmental Conditions: {env_sensor.get_conditions_string()}")
    
    # Run the main navigation loop
    main_navigation_loop()
