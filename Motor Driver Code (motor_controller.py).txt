*This Python class provides a clean interface to control the motors.*

# motor_controller.py
import RPi.GPIO as GPIO
import time

# Use Broadcom SOC Channel numbering (GPIO numbers, not pin numbers)
GPIO.setmode(GPIO.BCM)
GPIO.setwarnings(False)  # Disable warnings

class MotorController:
    def __init__(self, ena_pin=18, in1_pin=17, in2_pin=27, enb_pin=24, in3_pin=22, in4_pin=23):
        # Initialize GPIO pins for Motor A (Right)
        self.ena = ena_pin
        self.in1 = in1_pin
        self.in2 = in2_pin
        
        # Initialize GPIO pins for Motor B (Left)
        self.enb = enb_pin
        self.in3 = in3_pin
        self.in4 = in4_pin

        # Setup all pins as OUTPUT
        GPIO.setup(self.ena, GPIO.OUT)
        GPIO.setup(self.in1, GPIO.OUT)
        GPIO.setup(self.in2, GPIO.OUT)
        GPIO.setup(self.enb, GPIO.OUT)
        GPIO.setup(self.in3, GPIO.OUT)
        GPIO.setup(self.in4, GPIO.OUT)
        
        # Initialize PWM instances for speed control
        self.pwm_a = GPIO.PWM(self.ena, 1000) # 1000 Hz frequency
        self.pwm_b = GPIO.PWM(self.enb, 1000)
        self.pwm_a.start(0) # Start with 0% duty cycle (stopped)
        self.pwm_b.start(0)
        
        # Initial state: motors stopped
        self.stop()

    def move_forward(self, speed=50):
        """Drive both motors forward at the given speed (0-100%)."""
        print(f"Moving Forward at speed {speed}")

        # Right Motor Forward
        GPIO.output(self.in1, GPIO.HIGH)
        GPIO.output(self.in2, GPIO.LOW)
        # Left Motor Forward
        GPIO.output(self.in3, GPIO.HIGH)
        GPIO.output(self.in4, GPIO.LOW)
        # Set speed
        self.pwm_a.ChangeDutyCycle(speed)
        self.pwm_b.ChangeDutyCycle(speed)

    def move_backward(self, speed=50):
        """Drive both motors backward at the given speed (0-100%)."""
        print(f"Moving Backward at speed {speed}")

        # Right Motor Backward
        GPIO.output(self.in1, GPIO.LOW)
        GPIO.output(self.in2, GPIO.HIGH)
        # Left Motor Backward
        GPIO.output(self.in3, GPIO.LOW)
        GPIO.output(self.in4, GPIO.HIGH)
        # Set speed
        self.pwm_a.ChangeDutyCycle(speed)
        self.pwm_b.ChangeDutyCycle(speed)

    def turn_left(self, speed=50):
        """Turn left by running the right motor forward and stopping the left."""
        print(f"Turning Left at speed {speed}")

        # Right Motor Forward
        GPIO.output(self.in1, GPIO.HIGH)
        GPIO.output(self.in2, GPIO.LOW)
        # Left Motor Stop
        GPIO.output(self.in3, GPIO.LOW)
        GPIO.output(self.in4, GPIO.LOW)
        # Set speed for right motor only
        self.pwm_a.ChangeDutyCycle(speed)
        self.pwm_b.ChangeDutyCycle(0)

    def turn_right(self, speed=50):
        """Turn right by running the left motor forward and stopping the right."""
        print(f"Turning Right at speed {speed}")

        # Right Motor Stop
        GPIO.output(self.in1, GPIO.LOW)
        GPIO.output(self.in2, GPIO.LOW)
        # Left Motor Forward
        GPIO.output(self.in3, GPIO.HIGH)
        GPIO.output(self.in4, GPIO.LOW)
        # Set speed for left motor only
        self.pwm_a.ChangeDutyCycle(0)
        self.pwm_b.ChangeDutyCycle(speed)

    def stop(self):
        """Stop both motors abruptly."""
        print("Stopping Motors")
        GPIO.output(self.in1, GPIO.LOW)
        GPIO.output(self.in2, GPIO.LOW)
        GPIO.output(self.in3, GPIO.LOW)
        GPIO.output(self.in4, GPIO.LOW)
        self.pwm_a.ChangeDutyCycle(0)
        self.pwm_b.ChangeDutyCycle(0)

    def cleanup(self):
        """Clean up GPIO resources. Call this when exiting the program."""
        print("Cleaning up GPIO")
        self.stop()
        self.pwm_a.stop()
        self.pwm_b.stop()
        GPIO.cleanup()

# Simple test function
if __name__ == "__main__":
    robot = MotorController()
    try:
        robot.move_forward(50)
        time.sleep(2)
        robot.turn_right(40)
        time.sleep(1)
        robot.move_backward(60)
        time.sleep(2)
        robot.turn_left(40)
        time.sleep(1)
    except KeyboardInterrupt:
        pass
    finally:
        robot.cleanup()
