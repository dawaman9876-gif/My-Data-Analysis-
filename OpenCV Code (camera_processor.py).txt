*This module captures images and uses OpenCV for simple larva detection.*

# camera_processor.py
import cv2
import numpy as np
from picamera2 import Picamera2
import time

class CameraProcessor:
    def __init__(self):
        print("Initializing Camera...")
        self.picam2 = Picamera2()
        # Configure a preview stream (lower resolution for faster processing)
        preview_config = self.picam2.create_preview_configuration(main={"size": (640, 480)})
        self.picam2.configure(preview_config)
        self.picam2.start()
        time.sleep(2)  # Let camera warm up
        print("Camera ready.")

    def capture_and_process_frame(self):
        """Captures a frame and performs simple 'larva detection'."""
        # Capture image to numpy array
        image_array = self.picam2.capture_array()
        
        # Convert RGB to BGR for OpenCV
        frame_bgr = cv2.cvtColor(image_array, cv2.COLOR_RGB2BGR)
        
        # SIMPLE DETECTION LOGIC (Replace with your trained ML model)
        # 1. Convert to HSV color space
        hsv = cv2.cvtColor(frame_bgr, cv2.COLOR_BGR2HSV)
        
        # 2. Define a range for dark, larva-like colors (this is a placeholder)
        lower_bound = np.array([0, 0, 0])
        upper_bound = np.array([180, 255, 80])
        
        # 3. Create a mask
        mask = cv2.inRange(hsv, lower_bound, upper_bound)
        
        # 4. Clean up the mask
        kernel = np.ones((5,5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        # 5. Find contours
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        larva_detected = False
        larva_count = 0
        
        for cnt in contours:
            area = cv2.contourArea(cnt)
            if area > 100:  # Filter out small noise
                larva_detected = True
                larva_count += 1
                # Draw a bounding box (for debugging/visualization)
                x, y, w, h = cv2.boundingRect(cnt)
                cv2.rectangle(frame_bgr, (x, y), (x+w, y+h), (0, 255, 0), 2)
                cv2.putText(frame_bgr, f"Larva", (x, y-10), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0,255,0), 2)
        
        # For a real application, you would return structured data, not the image.
        result_data = {
            "detection_status": larva_detected,
            "larva_count": larva_count,
            # "frame": frame_bgr  # Only include if you need to stream video
        }
        return result_data

    def cleanup(self):
        """Releases the camera resource."""
        self.picam2.stop()
        print("Camera stopped.")
