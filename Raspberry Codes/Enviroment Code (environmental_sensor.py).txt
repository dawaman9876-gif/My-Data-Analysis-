# environmental_sensor.py
import adafruit_dht
import board
import time
import logging

# Setup logging for this module
logger = logging.getLogger("EnvSensor")

class EnvironmentalSensor:
    def __init__(self, sensor_pin=board.D4):
        """
        Initialize the DHT22 sensor.
        :param sensor_pin: The GPIO pin the sensor data line is connected to (using BCM numbering)
        """
        print("Initializing DHT22 Environmental Sensor...")
        self.sensor = adafruit_dht.DHT22(sensor_pin)
        self.temperature = None
        self.humidity = None
        time.sleep(1.0)  # Let sensor stabilize
        print("DHT22 Sensor ready.")

    def read_data(self):
        """Attempt to read temperature and humidity from the sensor."""
        try:
            self.temperature = self.sensor.temperature
            self.humidity = self.sensor.humidity
            # Log successful reading
            logger.info(f"ENV_DATA - Temperature: {self.temperature:.1f}C, Humidity: {self.humidity:.1f}%")
            return True
            
        except RuntimeError as error:
            # This is common for DHT sensors - just log the error and continue
            logger.warning(f"ENV_DATA - Sensor read error: {error.args[0]}")
            return False
            
        except Exception as error:
            logger.error(f"ENV_DATA - Unexpected error: {error}")
            self.sensor.exit()
            raise error

    def get_temperature(self):
        """Get the last successful temperature reading."""
        return self.temperature

    def get_humidity(self):
        """Get the last successful humidity reading."""
        return self.humidity

    def get_conditions_string(self):
        """Return a formatted string of current conditions."""
        if self.temperature is not None and self.humidity is not None:
            return f"{self.temperature:.1f}C, {self.humidity:.1f}% RH"
        return "No data available"

    def cleanup(self):
        """Clean up the sensor resources."""
        print("Cleaning up Environmental Sensor...")
        self.sensor.exit()

# Simple test function
if __name__ == "__main__":
    import logging
    logging.basicConfig(level=logging.INFO)
    
    env_sensor = EnvironmentalSensor()
    try:
        for i in range(10):
            if env_sensor.read_data():
                print(f"Environment: {env_sensor.get_conditions_string()}")
            time.sleep(2)
    except KeyboardInterrupt:
        print("Test interrupted.")
    finally:
        env_sensor.cleanup()
