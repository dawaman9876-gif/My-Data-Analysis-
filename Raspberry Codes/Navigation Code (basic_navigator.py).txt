# basic_navigator.py
from motor_controller import MotorController
import time

class BasicNavigator:
    def __init__(self):
        self.motors = MotorController()

    def execute_mission(self, mission_time=30):
        """A simple example mission that runs for a set time."""
        print(f"Starting mission for {mission_time} seconds.")
        start_time = time.time()

        try:
            while (time.time() - start_time) < mission_time:
                # This is a placeholder for your main logic.
                # In a real system, you would check sensors here.
                # Example: If an obstacle is detected, turn. Else, go forward.

                # For now, we'll just drive forward in a straight line.
                self.motors.move_forward(50)
                time.sleep(0.1) # Small delay to prevent CPU overload

        except KeyboardInterrupt:
            print("Mission interrupted by user.")
        finally:
            self.motors.stop()
            print("Mission complete.")

    def avoid_obstacle(self):
        """A simple obstacle avoidance maneuver."""
        print("Obstacle detected! Avoiding...")
        # 1. Stop
        self.motors.stop()
        time.sleep(0.5)
        # 2. Move back a little
        self.motors.move_backward(40)
        time.sleep(1)
        # 3. Turn right ~90 degrees
        self.motors.turn_right(60)
        time.sleep(0.8) # Adjust this time for a 90-degree turn
        # 4. Continue forward
        self.motors.move_forward(50)

# Run the navigator
if __name__ == "__main__":
    robot = BasicNavigator()
    robot.execute_mission(mission_time=10) # Run a 10-second test mission
    robot.motors.cleanup()
