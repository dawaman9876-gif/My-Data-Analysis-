*This module handles distance measurement with an HC-SR04 sensor.*

# ultrasonic_sensor.py
import RPi.GPIO as GPIO
import time

class UltrasonicSensor:
    def __init__(self, trigger_pin, echo_pin):
        self.GPIO_TRIGGER = trigger_pin
        self.GPIO_ECHO = echo_pin
        
        GPIO.setmode(GPIO.BCM)
        GPIO.setup(self.GPIO_TRIGGER, GPIO.OUT)
        GPIO.setup(self.GPIO_ECHO, GPIO.IN)
        
        # Settle the sensor
        GPIO.output(self.GPIO_TRIGGER, GPIO.LOW)
        time.sleep(0.5)
        print(f"Ultrasonic Sensor initialized on TRIG: {trigger_pin}, ECHO: {echo_pin}")

    def get_distance(self):
        """Returns distance in centimeters."""
        # Send a 10us pulse to trigger
        GPIO.output(self.GPIO_TRIGGER, GPIO.HIGH)
        time.sleep(0.00001)  # 10 microseconds
        GPIO.output(self.GPIO_TRIGGER, GPIO.LOW)
        
        start_time = time.time()
        stop_time = time.time()
        
        # Save the time when the pulse is sent
        while GPIO.input(self.GPIO_ECHO) == 0:
            start_time = time.time()
            if start_time - stop_time > 0.1:  # Timeout prevention
                return -1
            
        # Save the time when the echo is received
        while GPIO.input(self.GPIO_ECHO) == 1:
            stop_time = time.time()
            if stop_time - start_time > 0.1:  # Timeout prevention
                return -1
            
        # Calculate distance (time difference * speed of sound / 2)
        time_elapsed = stop_time - start_time
        distance_cm = (time_elapsed * 34300) / 2  # 343 m/s = 34300 cm/s
        
        return round(distance_cm, 2)

    def cleanup(self):
        """Clean up GPIO pins used by the sensor."""
        GPIO.cleanup((self.GPIO_TRIGGER, self.GPIO_ECHO))

# Simple test
if __name__ == "__main__":
    sensor = UltrasonicSensor(trigger_pin=5, echo_pin=6)
    try:
        for i in range(10):
            dist = sensor.get_distance()
            print(f"Distance: {dist} cm")
            time.sleep(1)
    finally:
        sensor.cleanup()
